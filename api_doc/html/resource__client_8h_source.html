50  bool call(ServiceType& msg, FailureBehavior failure_behavior)
51  {
52  // store the internal id that is generated by resource manager
53  // this client's internal id is automatically added to corresponding server query, when the
54  // client was called due to server callback. Otherwise it is owners responsibility to store it.
55  temoto_id::ID internal_resource_id = msg.response.rmp.resource_id;
56 
57  // search for given service request from previous queries
58  auto q_it = std::find_if(queries_.begin(), queries_.end(),
59  [&](const ClientQuery<ServiceType, Owner>& q) -> bool {
60  return q.getMsg().request == msg.request && !q.failed_;
61  });
62  if (q_it == queries_.end())
63  {
64  // New request
65  // Prepare return topic
66  std::string temoto_namespace = common::getTemotoNamespace();
67  std::string topic = '/' + temoto_namespace + '/' + this->resource_manager_.getName() + "/status";
68  msg.request.rmp.status_topic = topic;
69  msg.request.rmp.temoto_namespace = temoto_namespace;
70 
71  TEMOTO_DEBUG("New query, performing external call to %s", service_client_.getService().c_str());
72 
73  if (service_client_.call(msg))
74  {
75  if (msg.response.rmp.code == status_codes::FAILED)
76  {
77  throw FORWARD_ERROR(msg.response.rmp.error_stack);
78  }
79  TEMOTO_DEBUG("Service call was sucessful. ext id: %ld", msg.response.rmp.resource_id);
80  queries_.emplace_back(msg, owner_);
81  q_it = std::prev(queries_.end()); // set iterator to the added query
82  }
83  else
84  {
85  throw CREATE_ERROR(error::Code::RMP_FAIL, "Service call to %s returned false.",
86  service_client_.getService().c_str());
87  }
88  }
89  else
90  {
91  // Equal request was found from stored list
92  TEMOTO_DEBUG("Existing request, using stored response");
93 
94  // Fill the response part with the already existing one from previous call.
95  msg.response = q_it->getMsg().response;
96  }
97 
98  // Update id in response part
99  msg.response.rmp.resource_id = internal_resource_id;
100 
101  q_it->addInternalResource(internal_resource_id, failure_behavior);
102 
103  return true;
104  }
