317  void unloadResource(temoto_core::UnloadResource::Request& req,
318  temoto_core::UnloadResource::Response& res)
319  {
320  // find first query that contains resource that should be unloaded
321  const temoto_id::ID ext_rid = req.resource_id;
322  waitForLock(queries_mutex_);
323  const auto found_query_it =
324  std::find_if(queries_.begin(), queries_.end(),
325  [ext_rid](const ServerQuery<ServiceType>& query) -> bool {
326  return query.hasExternalResource(ext_rid);
327  });
328  if (found_query_it != queries_.end())
329  {
330 
331  TEMOTO_DEBUG("Query with ext id %d was found", ext_rid);
332  TEMOTO_DEBUG("internal resource count: %lu", found_query_it->getLinkedResources().size());
333 
334  // Query found, try to remove external client from it.
335  size_t resources_left = found_query_it->removeExternalResource(ext_rid);
336  if (resources_left == 0)
337  {
338  // last resource removed, execute owner's unload callback and remove the query from our list
339  typename ServiceType::Request orig_req = found_query_it->getMsg().request;
340  typename ServiceType::Response orig_res = found_query_it->getMsg().response;
341  error::ErrorStack unload_errs; // buffer for all unload-related errors
342  try
343  {
344  (owner_->*unload_callback_)(orig_req, orig_res);
345  }
346  catch(error::ErrorStack& error_stack)
347  {
348  unload_errs += error_stack;
349  }
350 
351  // Send unload command to all linked internal clients...
352  for (auto& set_el : found_query_it->getLinkedResources())
353  {
354  try
355  {
356  this->resource_manager_.unloadClientResource(set_el);
357  }
358  catch (error::ErrorStack& es)
359  {
360  unload_errs += es; //append error to the unload error stack
361  }
362  }
363 
364  // forward the stack if any error occured
365  if (unload_errs.size())
366  {
367  res.code = status_codes::FAILED;
368  res.error_stack += FORWARD_ERROR(unload_errs);
369  }
370 
371  // Finally, remove the found query, even when some of the unload calls failed.
372  // \TODO: The next line potentially causes zombie resources. How to manage these?
373  queries_.erase(found_query_it);
374  }
375  }
376  queries_mutex_.unlock();
377  }
