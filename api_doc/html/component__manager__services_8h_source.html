54 static bool operator==(const temoto_component_manager::LoadComponent::Request& r1,
55  const temoto_component_manager::LoadComponent::Request& r2)
56 {
57  // First check the basics
58  if (r1.component_type != r2.component_type ||
59  r1.package_name != r2.package_name ||
60  r1.executable != r2.executable)
61  {
62  return false;
63  }
64 
65  // Check the size of the requested output topics
66  if (r1.output_topics.size() != r2.output_topics.size())
67  {
68  return false;
69  }
70 
71  // Check the requested output topics
72  auto output_topics_2_copy = r2.output_topics;
73  for (auto& output_topic_1 : r1.output_topics)
74  {
75  bool topic_found = false;
76  for (auto it=output_topics_2_copy.begin(); it!=output_topics_2_copy.end(); it++)
77  {
78  if (output_topic_1.key == it->key &&
79  output_topic_1.value == it->value)
80  {
81  topic_found = true;
82  output_topics_2_copy.erase(it);
83  break;
84  }
85  }
86 
87  if (!topic_found)
88  {
89  return false;
90  }
91  }
92 
93  // The component infos are equal
94  return true;
95 }

105 static bool operator==(const temoto_component_manager::LoadPipe::Request& r1,
