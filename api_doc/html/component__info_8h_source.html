155 static bool operator==(const ComponentInfo& s1, const ComponentInfo& s2)
156 {
157  // Check the namespace, executable and name of the package
158  if (s1.getTemotoNamespace() != s2.getTemotoNamespace() ||
159  s1.getExecutable() != s2.getExecutable() ||
160  s1.getPackageName() != s2.getPackageName())
161  {
162  return false;
163  }
164 
165  // Check the size of input and output topics
166  if (s1.getInputTopics().size() != s2.getInputTopics().size() ||
167  s1.getOutputTopics().size() != s2.getOutputTopics().size())
168  {
169  return false;
170  }
171 
172  // Check the input topics
173  auto input_topics_2_copy = s2.getInputTopics();
174  for (auto& input_topic_1 : s1.getInputTopics())
175  {
176  bool topic_found = false;
177  for (auto it=input_topics_2_copy.begin(); it!=input_topics_2_copy.end(); it++)
178  {
179  if (input_topic_1.first == it->first)
180  {
181  topic_found = true;
182  input_topics_2_copy.erase(it);
183  break;
184  }
185  }
186 
187  if (!topic_found)
188  {
189  return false;
190  }
191  }
192 
193  // Check the output topics
194  auto output_topics_2_copy = s2.getOutputTopics();
195  for (auto& output_topic_1 : s1.getOutputTopics())
196  {
197  bool topic_found = false;
198  for (auto it=output_topics_2_copy.begin(); it!=output_topics_2_copy.end(); it++)
199  {
200  if (output_topic_1.first == it->first)
201  {
202  topic_found = true;
203  output_topics_2_copy.erase(it);
204  break;
205  }
206  }
207 
208  if (!topic_found)
209  {
210  return false;
211  }
212  }
213 
214  // The component infos are equal
215  return true;
216 }
