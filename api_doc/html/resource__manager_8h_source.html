108  template <class ServiceType>
109  void call(std::string resource_manager_name, std::string server_name, ServiceType& msg,
110  rmp::FailureBehavior failure_behavior = rmp::FailureBehavior::NONE,
111  std::string temoto_namespace = ::temoto_core::common::getTemotoNamespace())
112  {
113  using ClientType = ResourceClient<ServiceType, Owner>;
114  using ClientPtr = std::shared_ptr<ClientType>;
115  using BaseClientPtr = BaseResourceClientPtr<Owner>;
116 
117  ClientPtr client_ptr = NULL;
118 
119  waitForLock(clients_mutex_);
120 
121  // check if this client already exists
122  std::string client_name = '/'+temoto_namespace + "/" + resource_manager_name + "/" + server_name;
123  typename std::vector<BaseClientPtr>::iterator client_it =
124  std::find_if(clients_.begin(), clients_.end(),
125  [&](const BaseClientPtr& c) -> bool { return c->getName() == client_name; });
126  if (client_it == clients_.end())
127  {
128  TEMOTO_DEBUG("Creating new resource client '%s'.", client_name.c_str());
129 
130  client_ptr = std::make_shared<ClientType>(temoto_namespace, resource_manager_name,
131  server_name, owner_, *this);
132  // Push to clients and convert to BaseResourceClient type
133  clients_.push_back(client_ptr);
134  client_it = std::prev(clients_.end()); // set iterator to the client pointer we just added
135  }
136  else
137  {
138  client_ptr = std::dynamic_pointer_cast<ClientType>(*client_it);
139  if (!client_ptr)
140  {
141  // cast failed
142  clients_mutex_.unlock();
143  throw CREATE_ERROR(error::Code::RMP_FAIL, "Dynamic Cast failed, last time the same service was called using different type?");
144  }
145  }
146 
147  // generate new id for owner as the call was not initiated from any servers callback
148  msg.response.rmp.resource_id = generateID();
149 
150  try
151  {
152  // make the call
153  client_ptr->call(msg, failure_behavior);
154 
155  if (active_server_)
156  {
157  // if call was sucessful and the call was initiated from server callback,
158  // link the client to the server
159  TEMOTO_DEBUG("Linking internal client.");
160  active_server_->linkInternalResource(msg.response.rmp.resource_id);
161  }
162  }
163  catch (error::ErrorStack& error_stack)
164  {
165  clients_.erase(client_it);
166  clients_mutex_.unlock();
167  throw FORWARD_ERROR(error_stack);
168  }
169 
170  clients_mutex_.unlock();
171  }
