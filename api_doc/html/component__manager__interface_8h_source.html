175  ComponentTopicsRes startComponent( const std::string& component_type
176  , const std::string& package_name
177  , const std::string& ros_program_name
178  , const ComponentTopicsReq& topics
179  , const ComponentTopicsReq& parameters
180  , bool use_only_local_components = false)
181  {
182  // Fill out the "StartComponentRequest" request
183  temoto_component_manager::LoadComponent srv_msg;
184  srv_msg.request.component_type = component_type;
185  srv_msg.request.package_name = package_name;
186  srv_msg.request.executable = ros_program_name;
187  srv_msg.request.use_only_local_components = use_only_local_components;
188  srv_msg.request.output_topics = topics.outputTopicsAsKeyValues();
189  srv_msg.request.input_topics = topics.inputTopicsAsKeyValues();
190  srv_msg.request.required_parameters = parameters.outputTopicsAsKeyValues();
191 
192  // Call the server
193  try
194  {
195  resource_manager_->template call<LoadComponent>(srv_name::MANAGER,
196  srv_name::SERVER,
197  srv_msg);
198  }
199  catch(temoto_core::error::ErrorStack& error_stack)
200  {
201  throw FORWARD_ERROR(error_stack);
202  }
203 
204  allocated_components_.push_back(srv_msg);
205  ComponentTopicsRes responded_topics;
206  responded_topics.setOutputTopicsByKeyValue( srv_msg.response.output_topics );
207 
208  return responded_topics;
209  }

219  {

226  throw FORWARD_ERROR(error_stack);

228 
229  // Find all instances where request part matches of what was given and unload each resource
230  temoto_component_manager::LoadComponent::Request req;
231  req.component_type = component_type;
232  req.package_name = package_name;
233  req.executable = ros_program_name;
234 
235  // The == operator used in the lambda function is defined in
236  // component manager services header
237  auto found_component_it = std::find_if(
238  allocated_components_.begin(),
239  allocated_components_.end(),
240  [&](const LoadComponent& srv_msg) -> bool{ return srv_msg.request == req; });
241 
242  if (found_component_it == allocated_components_.end())
243  {
244  throw CREATE_ERROR(temoto_core::error::Code::RESOURCE_UNLOAD_FAIL, "Unable to unload resource that is not "
245  "loaded.");

247

249  {
250  // do the unloading
251  resource_manager_->unloadClientResource(found_component_it->response.rmp.resource_id);

414  void statusInfoCb(temoto_core::ResourceStatus& srv)
415  {
416  try
417  {
418  validateInterface();
419 
420  TEMOTO_INFO_STREAM("status info was received");
421  TEMOTO_INFO_STREAM(srv.request);
422 
423  /*
424  * If the status message indicates a resource failure, then find out what was
425  * the exact resource and execute a recovery behaviour
426  */
427  if (srv.request.status_code == temoto_core::rmp::status_codes::FAILED)
428  {
429  TEMOTO_WARN("The status info reported a resource failure.");
430 
431  /*
432  * Check if the resource that failed was a component
433  */
434  auto component_it = std::find_if(
435  allocated_components_.begin(),
436  allocated_components_.end(),
437  [&](const temoto_component_manager::LoadComponent& comp) -> bool {
438  return comp.response.rmp.resource_id == srv.request.resource_id;
439  });
440 
441  if (component_it != allocated_components_.end())
442  {
443  TEMOTO_DEBUG("Sending a request to unload the failed component ...");
444  resource_manager_->unloadClientResource(component_it->response.rmp.resource_id);
445 
446  /*
447  * Check if the owner action has a status routine defined
448  */
449  if (component_status_callback_)
450  {
451  TEMOTO_WARN_STREAM("Executing a custom recovery behaviour defined in Action '"
452  << owner_instance_->getName() << "'.");
453  (owner_instance_->*component_status_callback_)(*component_it);
454  return;
455  }
456  else
457  {
458  // Execute the default behavior for component failure, which is to load a new component
459  TEMOTO_DEBUG("Asking the same component again");
460 
461  // this call automatically updates the response in allocated components vec
462  component_it->request.output_topics = component_it->response.output_topics;
463  resource_manager_->template call<LoadComponent>(srv_name::MANAGER,
464  srv_name::SERVER,
465  *component_it);
466  }
467  return;
468  }
469 
470  /*
471  * Check if the resource that failed was a pipe
472  */
473  auto pipe_it = std::find_if(
474  allocated_pipes_.begin(),
475  allocated_pipes_.end(),
476  [&](const LoadPipe& pipe) -> bool {
477  return pipe.response.rmp.resource_id == srv.request.resource_id;
478  });
479 
480  if (pipe_it != allocated_pipes_.end())
481  {
482  TEMOTO_DEBUG("Sending a request to unload the failed pipe ...");
483  resource_manager_->unloadClientResource(component_it->response.rmp.resource_id);
484 
485  /*
486  * Check if the owner action has a status routine defined
487  */
488  if (pipe_status_callback_)
489  {
490  TEMOTO_WARN_STREAM("Executing a custom recovery behaviour defined in Action '"
491  << owner_instance_->getName() << "'.");
492  (owner_instance_->*pipe_status_callback_)(*pipe_it);
493  return;
494  }
495  else
496  {
497  // ... copy the output topics from the response into the output topics
498  // of the request (since the user still wants to receive the data on the same topics) ...
499  pipe_it->request.output_topics = pipe_it->response.output_topics;
500  pipe_it->request.pipe_id = pipe_it->response.pipe_id;
501 
502  // ... and load an alternative pipe. This call automatically
503  // updates the response in allocated pipes vector
504  TEMOTO_DEBUG_STREAM("Trying to load an alternative pipe");
505  resource_manager_->template call<LoadPipe>(srv_name::MANAGER_2,
506  srv_name::PIPE_SERVER,
507  *pipe_it);
508  }
509  return;
510  }
511 
512  TEMOTO_ERROR_STREAM("Resource status arrived for a resource that does not exist.");
513  // throw CREATE_ERROR(temoto_core::error::Code::RESOURCE_NOT_FOUND, "Resource status arrived for a "
514  // "resource that does not exist.");
515  }
516  }
517  catch (temoto_core::error::ErrorStack& error_stack)
518  {
519  throw FORWARD_ERROR(error_stack);
520  }
521  }
